// Image cache.
const imageCache = new Map();

/**
 * Returns the image associated with a brawler name.
 * @param {string} brawlerName - The brawler name.
 * @returns {*} The resolved image asset, or null when not found.
 */
export const getBrawlerImage = (brawlerName) => {
  if (!brawlerName) {
    return null;
  }

  // Check the cache first.
  if (imageCache.has(brawlerName)) {
    return imageCache.get(brawlerName);
  }

  // Resolve the image asset.
  const image = brawlerImagesMap[brawlerName] || null;

  // Store the image in the cache.
  if (image) {
    imageCache.set(brawlerName, image);
  }

  return image;
};
